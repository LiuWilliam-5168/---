const int maxn = (int)22;
const int MOD = (int)1e9 + 7;

int n, m;
vector<int> e[maxn];
int dp[maxn][1 << maxn];

int f(int i, int mask) {
    if (i == 0)
        return dp[i][mask] = (mask == 1) ? 1 : 0;

    if (!(mask & (1 << i)))
        return 0;

    if (dp[i][mask]) 
        return dp[i][mask];
    
    int rtn = 0;
    for (int j : e[i]) {
        if (mask & (1 << j)) {
            rtn += f(j, mask - (1 << i));
            if (rtn >= MOD) rtn -= MOD;
        }
    }
    return dp[i][mask] = rtn;
}

void solve() {
    cin >> n >> m;

    for (int i = 1; i <= m; i++) {
        int u, v;  cin >> u >> v;
        e[v - 1].push_back(u - 1);
    }

    cout << f(n - 1, (1 << n) - 1) << '\n';
}
